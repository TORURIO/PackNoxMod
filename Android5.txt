package com.example.checkboxradiobutton;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.CheckBox;
import android.widget.RadioButton;
import android.widget.RadioGroup;
import android.widget.Toast;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    CheckBox checkBoxKemBo, checkBoxKemSauRieng, checkBoxKemDau;
    RadioGroup radioGroupSize, radioGroupMath;
    Button btnSubmit;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        checkBoxKemBo = findViewById(R.id.checkBoxKemBo);
        checkBoxKemSauRieng = findViewById(R.id.checkBoxKemSauRieng);
        checkBoxKemDau = findViewById(R.id.checkBoxKemDau);
        radioGroupSize = findViewById(R.id.radioGroupSize);
        radioGroupMath = findViewById(R.id.radioGroupMath);
        btnSubmit = findViewById(R.id.btnSubmit);

        btnSubmit.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // Kiểm tra chọn loại kem
                String selectedKem = "";
                if (checkBoxKemBo.isChecked()) {
                    selectedKem += "Kem bơ ";
                }
                if (checkBoxKemSauRieng.isChecked()) {
                    selectedKem += "Kem sầu riêng ";
                }
                if (checkBoxKemDau.isChecked()) {
                    selectedKem += "Kem dâu ";
                }
                if (selectedKem.isEmpty()) {
                    Toast.makeText(MainActivity.this, "Bạn phải chọn ít nhất một loại kem", Toast.LENGTH_SHORT).show();
                    return;
                }

                // Kiểm tra chọn size
                int selectedSizeId = radioGroupSize.getCheckedRadioButtonId();
                if (selectedSizeId == -1) {
                    Toast.makeText(MainActivity.this, "Bạn phải chọn size", Toast.LENGTH_SHORT).show();
                    return;
                }

                RadioButton selectedSize = findViewById(selectedSizeId);

                // Kiểm tra chọn phép tính 1 + 1
                int selectedMathId = radioGroupMath.getCheckedRadioButtonId();
                if (selectedMathId == -1) {
                    Toast.makeText(MainActivity.this, "Bạn phải chọn đáp án cho 1 + 1", Toast.LENGTH_SHORT).show();
                    return;
                }

                RadioButton selectedMath = findViewById(selectedMathId);
                if (selectedMath.getText().toString().equals("2")) {
                    Toast.makeText(MainActivity.this, "Bạn đã chọn: " + selectedKem + "Size: " + selectedSize.getText() + "\nĐáp án 1 + 1 = 2 là đúng!", Toast.LENGTH_SHORT).show();
                } else {
                    Toast.makeText(MainActivity.this, "Bạn đã chọn sai đáp án cho 1 + 1!", Toast.LENGTH_SHORT).show();
                }
            }
        });
    }
}

package com.example.radiobuttontest

import android.os.Bundle
import android.widget.*
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    lateinit var radioGroup: RadioGroup
    lateinit var submitButton: Button
    lateinit var selectedOptionTextView: TextView

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        radioGroup = findViewById(R.id.radioGroup)
        submitButton = findViewById(R.id.submitButton)
        selectedOptionTextView = findViewById(R.id.selectedOptionTextView)

        submitButton.setOnClickListener {
            val selectedId = radioGroup.checkedRadioButtonId
            val radioButton = findViewById<RadioButton>(selectedId)
            selectedOptionTextView.text = "Selected Option: ${radioButton.text}"
        }
    }
}
